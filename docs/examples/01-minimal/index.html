Example 01 — Minimal Pulse app

Shows the smallest useful Pulse app with middleware, GET, POST, and an error boundary.

Run this shape in any fetch-style runtime with app.bind().

01-minimal/index.ts
import { Pulse, Router } from 'pulse';

const pulse = new Pulse();
const app = new Router({ pulse });

app
  .use(async (ctx, next) => {
    ctx.state.startedAt = Date.now();
    return next();
  })
  .get('/health', async (ctx) => {
    return ctx.reply({ ok: true, startedAt: ctx.state.startedAt });
  })
  .post('/compute/:id', async (ctx) => {
    return ctx.reply({ ok: true, id: ctx.params.id });
  })
  .error(async (err, ctx) => {
    return ctx.reply({ ok: false, error: String(err) }, { status: 500 });
  });

export default app.bind();
