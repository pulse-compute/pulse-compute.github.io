Example 04 — GRIP gateway usage

Shows pulse.grip as an additive gateway capability.

This example assumes a GRIP-compatible gateway such as Pushpin or Fastly Fanout.

04-grip-gateway/index.ts
import { Pulse, Router } from 'pulse';
import { GripGateway } from 'pulse.grip';

const grip = new GripGateway({
  grip: process.env.GRIP_URL,
  ctxName: 'grip'
});

const pulse = new Pulse({
  runtime: {
    gateway: grip,
    broadcaster: GripGateway.broadcaster({ grip: process.env.GRIP_URL }),
    verify: 'required'
  }
});

const app = new Router({ pulse });

app
  .use(grip)
  .get('/realtime/:id', async (ctx) => {
    return ctx.grip!.subscribe(`bundle:${ctx.params.id}`, {
      json: { ready: true }
    });
  })
  .post('/broadcast/:id', async (ctx) => {
    await pulse.broadcast(`bundle:${ctx.params.id}`, {
      type: 'delta',
      ok: true
    });
    return ctx.reply({ broadcast: true });
  });

export default app.bind();
